85 inline AABB post_transform(const glm::vec3& T, const glm::mat3& R) const
86 {
87 AABB aabb;
88
89 // For all three axes
90 for (int i = 0; i < 3; i++)
91 {
92 aabb.min[i] = aabb.max[i] = T[i];
93 for (int j = 0; j < 3; j++)
94 {
95 float e = R[j][i] * min[j];
96 float f = R[j][i] * max[j];
97 if (e < f)
98 {
99 aabb.min[i] += e;
100 aabb.max[i] += f;
101 }
102 else
103 {
104 aabb.min[i] += f;
105 aabb.max[i] += e;
106 }
107 }
108 }
109
110 return aabb;
111 }

113 inline AABB post_transform(const glm::mat4& M) const
114 {
115 AABB aabb;
116
117 // For all three axes
118 for (int i = 0; i < 3; i++) {
119 // Start by adding in translation
120 aabb.min[i] = aabb.max[i] = M[3][i];
121 // Form extent by summing smaller and larger terms respectively
122 for (int j = 0; j < 3; j++) {
123 float e = M[j][i] * min[j];
124 float f = M[j][i] * max[j];
125 if (e < f) {
126 aabb.min[i] += e;
127 aabb.max[i] += f;
128 }
129 else {
130 aabb.min[i] += f;
131 aabb.max[i] += e;
132 }
133 }
134 }
135 return aabb;
136 }

146 {

152 return false;
