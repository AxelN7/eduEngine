57 bool insert(NodeType node, const std::string &parent_name)
58 {
59 // No parent given - insert as root
60 if (!parent_name.size())
61 {
62 nodes.insert(nodes.begin(), node);
63 return true;
64 }
65
66 // Locate parent
67 auto parent_index = find_node_index(parent_name);
68 if (parent_index == EENG_NULL_INDEX)
69 return false;
70 auto pit = nodes.begin() + parent_index;
71
72 // Update branch strides within the same branch
73 // Iterate backwards from parent to the root and increment strides
74 // ranging to the insertion
75 //
76 auto prit = pit;
77 while (prit >= nodes.begin())
78 {
79 if (prit->m_branch_stride > (unsigned)std::distance(prit, pit))
80 prit->m_branch_stride++;
81 if (!prit->m_parent_ofs) // Discontinue past root (preceeding trees not affected)
82 break;
83 prit--;
84 }
85
86 // Update parent indices
87 // Iterate forward up until the next root,
88 // and increment parent offsets ranging to the insertion
89 //
90 auto pfit = pit + 1;
91 while (pfit < nodes.end())
92 {
93 if (!pfit->m_parent_ofs) // Discontinue at succeeding root
94 break;
95 if (pfit->m_parent_ofs >= (unsigned)std::distance(pit, pfit))
96 pfit->m_parent_ofs++;
97 pfit++;
98 }
99
100 // Insert new node after its parent
101 node.m_parent_ofs = 1;
102 pit->m_nbr_children++;
103 nodes.insert(pit + 1, node);
104
105 return true;
106 }
